CMAKE_MINIMUM_REQUIRED(VERSION 3.8)
PROJECT(CppJSON)

INCLUDE_DIRECTORIES(
    ${PROJECT_SOURCE_DIR}/include
)

AUX_SOURCE_DIRECTORY(
    ${PROJECT_SOURCE_DIR}/src
    SOURCES
)

ADD_LIBRARY(
    ${PROJECT_NAME}
    ${SOURCES}
)


TARGET_INCLUDE_DIRECTORIES(${PROJECT_NAME} 
    PUBLIC
    ${PROJECT_SOURCE_DIR}/include
)
SET(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/../bin)
